使用数组模拟栈结构代码实现

package com.java.stack;

import java.util.Scanner;

/**
 * 使用数组来模拟栈结构
 */
public class ArrayStackDemo {
    public static void main(String[] args) {
        StackArr stack = new StackArr(4);
        boolean loop = true;
        Scanner scan = new Scanner(System.in);
        char key = ' ';    
        while(loop) {
            System.out.println("e(exit) 退出");
            System.out.println("a(add) 入栈操作");
            System.out.println("r(remove) 出栈操作");
            System.out.println("t(top) 查看当前栈顶的元素");
            key = scan.next().charAt(0);
            
            switch(key) {
            case 'e':
                System.out.println("**********退出程序**********");
                loop = false;
                break;
            case 'a':
                System.out.println("**********请输入要入栈的数据**********");
                int val = scan.nextInt();
                stack.push(val);
                break;
            case 'r':
                System.out.println("**********出栈一个数据**********");
                int pop = stack.pop();
                System.out.println("出栈的元素是："+pop);
                break;
            case 't':
                System.out.println("**********查看当前的栈顶元素**********");
                int top = stack.top();
                System.out.println("当前栈顶的元素是："+top);
                break;
            default:
                System.out.println("没有这条指令，重新输入");
                break;        
            }
        }
    }
}

//栈的一个结构
class StackArr{    
    int[] arr = null;   //使用数组模拟栈结构
    int maxSize;        //栈的最大容量
    int top;            //栈顶指针    
    public StackArr(int maxSize) {
        this.maxSize = maxSize;
        arr = new int[maxSize];
        top = -1;
    }
    //空栈
    public boolean isEmpty() {
        return top==-1?true:false;
    }
    //满栈
    public boolean isFull() {
        return top==maxSize-1?true:false;
    }
    //入栈操作
    public void push(int n) {
        if(isFull()) {
            System.out.println("满栈，无法加入~~");
            return;
        }
        arr[++top] = n; 
    }
    //出栈操作
    public int pop() {    
        if(isEmpty()) {
            throw new RuntimeException("空栈，无法进行pop操作");
        }
        int value = arr[top--];
        return value;
    }
    //查看栈顶元素
    public int top() {
        if(isEmpty()) {
            throw new RuntimeException("空栈，栈顶为空");
        }
        return arr[top];
    }
}
